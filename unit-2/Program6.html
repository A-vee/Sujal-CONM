6. Program of Gauss Jordan.

   #include < stdio.h>
        
   int main() {
       int i, j, k, n;
       float A[20][20], c, x[10];
   
       printf("Enter the size of matrix: ");
       scanf("%d", &n);
   
       printf("Enter the elements of augmented matrix row-wise:\n");
       for (i = 1; i <= n; i++) {
           for (j = 1; j <= n + 1; j++) {
               printf("A[%d][%d]: ", i, j);
               scanf("%f", &A[i][j]);
           }
       }
   
       // Applying Gauss-Jordan Elimination
       for (j = 1; j <= n; j++) {
           for (i = 1; i <= n; i++) {
               if (i != j) {
                   c = A[i][j] / A[j][j];
                   for (k = 1; k <= n + 1; k++) {
                       A[i][k] = A[i][k] - c * A[j][k];
                   }
               }
           }
       }
   
       printf("The solution is:\n");
       for (i = 1; i <= n; i++) {
           x[i] = A[i][n + 1] / A[i][i];
           printf("x%d = %f\n", i, x[i]);
       }
   
       return 0;
   }
