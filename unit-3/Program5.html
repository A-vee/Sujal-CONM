5.Program of Newton's Divided Difference

   #include < stdio.h>

           
   void dividedDifference(double x[], double y[], double coef[], int n) {
           for (int i = 0; i < n; i++) {
               coef[i] = y[i];
           }
           for (int j = 1; j < n; j++) {
               for (int i = n - 1; i >= j; i--) {
                   coef[i] = (coef[i] - coef[i - 1]) / (x[i] - x[i - j]);
               }
           }
       }
       
       
       double interpolate(double x[], double coef[], int n, double value) {
           double result = coef[n - 1];
           for (int i = n - 2; i >= 0; i--) {
               result = result * (value - x[i]) + coef[i];
           }
           return result;
       }
       
       int main() {
           int n;
           printf("Enter the number of data points: ");
           scanf("%d", &n);
       
           double x[n], y[n], coef[n];
           printf("Enter the x values:\n");
           for (int i = 0; i < n; i++) {
               scanf("%lf", &x[i]);
           }
           printf("Enter the y values:\n");
           for (int i = 0; i < n; i++) {
               scanf("%lf", &y[i]);
           }
       
           dividedDifference(x, y, coef, n);
       
           double value;
           printf("Enter the value to interpolate: ");
           scanf("%lf", &value);
       
           double result = interpolate(x, coef, n, value);
           printf("Interpolated value at %.2lf is %.2lf\n", value, result);
       
           return 0;
       }
