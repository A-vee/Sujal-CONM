1. Euler's method

   #include <studio.h>

   // Define the function f(x, y) = dy/dx
     double f(double x, double y) {
       return x + y; // Example: dy/dx = x + y
   }
   
   // Euler's method function
    void eulerMethod(double x0, double y0, double h, double xn) {
       double x = x0;
       double y = y0;
       
       printf("x\t\ty\n");
       printf("%.4f\t%.4f\n", x, y);
       
       while (x < xn) {
           y = y + h * f(x, y);  // Euler's formula: y_(n+1) = y_n + h*f(x_n, y_n)
           x = x + h;
           
           printf("%.4f\t%.4f\n", x, y); // Display x and y at each step
       }
   }
   
       int main() {
       double x0, y0, h, xn;
       
       // Input initial conditions and step size
       printf("Enter initial value of x (x0): ");
       scanf("%lf", &x0);
       printf("Enter initial value of y (y0): ");
       scanf("%lf", &y0);
       printf("Enter step size (h): ");
       scanf("%lf", &h);
       printf("Enter the value of x to find y (xn): ");
       scanf("%lf", &xn);
       
       // Perform Euler's method
       eulerMethod(x0, y0, h, xn);
       
       return 0;
   }
