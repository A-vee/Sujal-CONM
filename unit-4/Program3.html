3. Modified euler's method

    #include <stdio.h > 

    // Define the function f(x, y) = dy/dx
    double f(double x, double y) {
        return x + y; // Example: dy/dx = x + y
    }
    
    // Modified Euler's method function
    void modifiedEuler(double x0, double y0, double h, double xn) {
        double x = x0;
        double y = y0;
        double k1, k2;
        
        printf("Modified Euler's Method\n");
        printf("x\t\ty\n");
        printf("%.4f\t%.4f\n", x, y);
        
        while (x < xn) {
            k1 = h * f(x, y);
            double y_temp = y + k1; // Temporary y value at x + h
            k2 = h * f(x + h, y_temp);
            
            y = y + (k1 + k2) / 2;
            x = x + h;
            
            printf("%.4f\t%.4f\n", x, y); // Display x and y at each step
        }
    }
    
    int main() {
        double x0, y0, h, xn;
        
        // Input initial conditions and step size
        printf("Enter initial value of x (x0): ");
        scanf("%lf", &x0);
        printf("Enter initial value of y (y0): ");
        scanf("%lf", &y0);
        printf("Enter step size (h): ");
        scanf("%lf", &h);
        printf("Enter the value of x to find y (xn): ");
        scanf("%lf", &xn);
        
        // Perform Modified Euler's method
        modifiedEuler(x0, y0, h, xn);
        
        return 0;
    }
