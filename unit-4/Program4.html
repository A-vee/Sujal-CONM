4. Taylor method

   #include >stdio.h<

   // Define the first derivative function f(x, y) = dy/dx
   double f(double x, double y) {
       return x + y; // Example: dy/dx = x + y
   }
   
   // Define the second derivative function f2(x, y) = d^2y/dx^2
   double f2(double x, double y) {
       return 1 + f(x, y); // d^2y/dx^2 = 1 + dy/dx
   }
   
   // Taylor series method function
   void taylorSeries(double x0, double y0, double h, double xn) {
       double x = x0;
       double y = y0;
       
       printf("Taylor Series Method (2nd Order)\n");
       printf("x\t\ty\n");
       printf("%.4f\t%.4f\n", x, y);
       
       while (x < xn) {
           double dy = h * f(x, y);         // First-order term: h * dy/dx
           double d2y = (h * h / 2) * f2(x, y); // Second-order term: (h^2 / 2) * d^2y/dx^2
           
           y = y + dy + d2y;  // Update y using Taylor series up to the second order
           x = x + h;
           
           printf("%.4f\t%.4f\n", x, y); // Display x and y at each step
       }
   }
   
   int main() {
       double x0, y0, h, xn;
       
       // Input initial conditions and step size
       printf("Enter initial value of x (x0): ");
       scanf("%lf", &x0);
       printf("Enter initial value of y (y0): ");
       scanf("%lf", &y0);
       printf("Enter step size (h): ");
       scanf("%lf", &h);
       printf("Enter the value of x to find y (xn): ");
       scanf("%lf", &xn);
       
       // Perform Taylor Series method
       taylorSeries(x0, y0, h, xn);
       
       return 0;
   }
