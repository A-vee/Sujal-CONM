      # Input binary numbers as strings
      binary1 = "1101"  # Example binary number 1
      binary2 = "1011"  # Example binary number 2
      
      # Convert binary numbers to decimal
      decimal1 = int(binary1, 2)
      decimal2 = int(binary2, 2)
      
      # Add the decimal numbers
      decimal_sum = decimal1 + decimal2
      
      # Convert the result back to binary
      binary_sum = bin(decimal_sum).replace("0b", "")
      
      print(f"Sum of {binary1} and {binary2} is {binary_sum}")
