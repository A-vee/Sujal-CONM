      # Define the initial guesses
      x0 = 2
      x1 = 3
      
      # Set the precision
      tolerance = 0.0001
      max_iterations = 1000  # Limit to avoid endless loops
      
      # Example equation: f(x) = x^3 - 4*x - 9
      def f(x):
          return x**3 - 4*x - 9
      
      # Secant method loop
      iterations = 0
      while iterations < max_iterations:
          # Calculate f(x0) and f(x1)
          f_x0 = f(x0)
          f_x1 = f(x1)
          
          # Calculate the new approximation
          if f_x1 - f_x0 == 0:  # Prevent division by zero
              print("Division by zero encountered. Exiting.")
              break
          
          x2 = x1 - (f_x1 * (x1 - x0)) / (f_x1 - f_x0)
          
          # Calculate f(x2)
          f_x2 = f(x2)
          
          # Print the current approximation and f(x2)
          print(f"Iteration {iterations}: x2 = {x2}, f(x2) = {f_x2}")
          
          # Check if x2 is within the desired tolerance
          if abs(f_x2) < tolerance:
              break
          
          # Update x0 and x1 for the next iteration
          x0 = x1
          x1 = x2
          
          iterations += 1
      
      # Final result (root of the equation)
      if iterations < max_iterations:
          print(f"The root is approximately: {x2}")
      else:
          print("Reached the maximum number of iterations without convergence.")
